	PN	FREQUENZ
	ORG	08000H
;
	LD	A,15
	OUT	1
	RST	020H
	DB	2
	DB	'Zeit Takt :'
	DB	0A0H
	RST	020H
	DB	16
	LD	DE,(016H)
	RST	020H
	DB	3
	LD	(TIME),HL
	RST	020H
	DB	2
	DB	'Zeit Low  :'
	DB	0A0H
	RST	020H
	DB	16
	LD	DE,(016H)
	RST	020H
	DB	3
	LD	(LOW),HL
;
LOOP:	LD	A,080H
	OUT	0
	LD	HL,(TIME)
TIM1:	DEC	HL
	LD	A,H
	OR	L
	JRNZ	TIM1-#
;
	LD	A,0
	OUT	0
	LD	HL,(LOW)
TIM2:	DEC	HL
	LD	A,H
	OR	L
	JRNZ	TIM2-#
;
	JR	LOOP-#
TIME:	DA	0
LOW:	DA	0
	END
;
	END
